# 1 - Configura S1 como servidor VTP
vtp mode server
vtp domain UOC
vtp password cisco

# 2 - Configura S2 y S3 como clientes
vtp mode client
vtp domain UOC
vtp password cisco

# 3 - Configura la VLAN
# S1 (Servidor VTP)
conf t
vlan 10
name profesorado
vlan 20
name alumnos
vlan 30
name invitados


4 - Configura enlaces troncales
# S1
int fa0/2
switchport mode trunk
int fa0/1
switchport mode trunk

# S2
int fa0/4
switchport mode trunk

# S3
int fa0/1
switchport mode trunk

# 7.- Añade el cable necesario entre S2 y S3, y configura STP para que la red siga funcionando correctamente.
# S2
spanning-tree vlan 10,20,30 root primary
int fa0/5
switchport mode trunk

# S3
spanning-tree vlan 10,20,30 root secondary
int fa0/5
switchport mode trunk
